Schéma Case

C
switch (<variables> {
case <Valeur1>: <action1>; break;
case <Valeur2>: <action2>; break;
...
case <Valeurn>: <actionn>; break;
default: <action>; break;
}

#include <stdio.h>

void main()
{
int j;
printf("Donner un numero etre 1 et 7: ");
scanf("%d", &j);

switch(j)
{
case 1: { printf("Lundi \n"); break; }
case 2: { printf("Mardi \n"); break; }
case 3: { printf("Mercredi \n"); break; }
case 4: { printf("Jeudi \n"); break; }
case 5: { printf("Vendredi \n"); break; }
case 6: { printf("Samedi \n"); break; }
case 7: { printf("Dimanche \n"); break; }
default: { printf("Incorrect \n"); break; }
}
}
